PREFIX orkgc: <http://orkg.org/orkg/class/>
PREFIX orkgp: <http://orkg.org/orkg/predicate/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?cont ?model_lbl ?bench ?data ?evaluation ?metric_lbl ?value
WHERE {
  ?cont       orkgp:HAS_BENCHMARK ?benchmark;
              orkgp:HAS_MODEL     ?model.
  ?benchmark  orkgp:HAS_DATASET   ?dataset;
              orkgp:HAS_EVALUATION ?eval.
  ?eval       orkgp:HAS_VALUE     ?value;
              orkgp:HAS_METRIC    ?metric.
 

  ?benchmark rdfs:label ?bench. 
  ?dataset   rdfs:label ?data.
  ?eval      rdfs:label ?evaluation.
  ?metric    rdfs:label ?metric_lbl.
  ?model    rdfs:label ?model_lbl.
  
  
  FILTER(CONTAINS(LCASE(STR(?metric_lbl)), "f1 score"))
  FILTER(CONTAINS(LCASE(STR(?model_lbl)), "hneabp (bwne)"))
   
}     
    ORDER BY DESC(?value)
    LIMIT 1
